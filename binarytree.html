C - Binary Tree Data Structure

C - Binary Tree Implementation

A binary tree is a data structure where each node has at most two children. Below is a simple implementation in C: 

// C program to demonstrate a simple binary tree implementation
#include <stdio.h>
#include <stdlib.h>

typedef struct Node {
  int data;
  struct Node* left;
  struct Node* right;
} Node;

Node* createNode(int data) {
  Node* new_node = malloc(sizeof(Node));
  new_node->data = data;
  new_node->left = NULL;
  new_node->right = NULL;
  return new_node;
}

void inOrderTraversal(Node* root) {
  if (root != NULL) {
    inOrderTraversal(root->left);
    printf("%d ", root->data);
    inOrderTraversal(root->right);
  }
}

int main() {
  Node* root = createNode(1);
  root->left = createNode(2);
  root->right = createNode(3);

  inOrderTraversal(root);
  return 0;
}

C++ - Binary Tree Implementation

In C++, binary trees can be implemented using classes. Here’s a basic example: 

// C++ program to demonstrate a simple binary tree implementation
#include <iostream>

class Node {
public:
  int data;
  Node* left;
  Node* right;

  Node(int val) : data(val), left(NULL), right(NULL) {}
};

void inOrderTraversal(Node* root) {
  if (root != NULL) {
    inOrderTraversal(root->left);
    std::cout << root->data << " ";
    inOrderTraversal(root->right);
  }
}

int main() {
  Node* root = new Node(1);
  root->left = new Node(2);
  root->right = new Node(3);

  inOrderTraversal(root);
  return 0;
}

Python - Binary Tree Implementation

In Python, you can implement a binary tree using classes. Here’s a simple example: 

# Python program to demonstrate a simple binary tree implementation

class Node:
  def __init__(self, data):
    self.data = data
    self.left = None
    self.right = None

def inOrderTraversal(self):
  if self.left:
    self.left.inOrderTraversal()
  print(self.data, end=" ")
  if self.right:
    self.right.inOrderTraversal()

if __name__ == "__main__":
  root = Node(1)
  root.left = Node(2)
  root.right = Node(3)

  root.inOrderTraversal()
