C - Heap Data Structure

C - Heap Implementation

In C, a heap can be implemented using an array. Below is an example of a Max Heap in C: 

/* C code for Max Heap implementation */
#include <stdio.h>

void heapify(int arr[], int n, int i) {
    int largest = i;
    int left = 2 * i + 1;
    int right = 2 * i + 2;

    if (left < n && arr[left] > arr[largest]) {
        largest = left;
    }
    if (right < n && arr[right] > arr[largest]) {
        largest = right;
    }

    if (largest != i) {
        int swap = arr[i];
        arr[i] = arr[largest];
        arr[largest] = swap;
        heapify(arr, n, largest);
    }
}

void buildHeap(int arr[], int n) {
    for (int i = n / 2 - 1; i >= 0; i--) {
        heapify(arr, n, i);
    }
}

int main() {
    int arr[] = {4, 10, 3, 5, 1};
    int n = sizeof(arr) / sizeof(arr[0]);

    buildHeap(arr, n);

    for (int i = 0; i < n; i++) {
        printf("%d ", arr[i]);
    }
    return 0;
}

C++ - Heap Implementation

In C++, a heap can also be implemented using an array. Below is an example of a Max Heap in C++: 

/* C++ code for Max Heap implementation */
#include <iostream>
#include <vector>
#include <algorithm>

void heapify(std::vector<int> &arr, int n, int i) {
    int largest = i;
    int left = 2 * i + 1;
    int right = 2 * i + 2;

    if (left < n && arr[left] > arr[largest]) {
        largest = left;
    }
    if (right < n && arr[right] > arr[largest]) {
        largest = right;
    }

    if (largest != i) {
        std::swap(arr[i], arr[largest]);
        heapify(arr, n, largest);
    }
}

int main() {
    std::vector<int> arr = {4, 10, 3, 5, 1};
    int n = arr.size();

    for (int i = n / 2 - 1; i >= 0; i--) {
        heapify(arr, n, i);
    }

    for (int i = 0; i < n; i++) {
        std::cout << arr[i] << " ";
    }
    std::cout << std::endl;
    return 0;
}

Python - Heap Implementation

In Python, heaps are typically implemented using the `heapq` library. Below is an example of using a heap in Python: 

# Python code using heapq library
import heapq

def heapify(arr):
    heapq.heapify(arr)

def push(arr, value):
    heapq.heappush(arr, value)

def pop(arr):
    return heapq.heappop(arr)

def main():
    arr = [4, 10, 3, 5, 1]
    heapify(arr)

    print("Heap elements:", arr)

    push(arr, 2)
    print("After pushing 2:", arr)

    pop(arr)
    print("After popping the smallest element:", arr)

main()
