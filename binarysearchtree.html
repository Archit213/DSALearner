C - Binary Search Tree Implementation

A Binary Search Tree (BST) is a binary tree where the left child is smaller than the parent node, and the right child is greater. Hereâ€™s an example in C: 

// C program for Binary Search Tree (BST) implementation
#include <stdio.h>
#include <stdlib.h>

typedef struct Node {
  int data;
  struct Node* left;
  struct Node* right;
} Node;

Node* createNode(int data) {
  Node* new_node = malloc(sizeof(Node));
  new_node->data = data;
  new_node->left = NULL;
  new_node->right = NULL;
  return new_node;
}

Node* insert(Node* root, int key) {
  if (root == NULL)
    return createNode(key);

  if (key < root->data)
    root->left = insert(root->left, key);
  else if (key > root->data)
    root->right = insert(root->right, key);

  return root;
}

void inOrderTraversal(Node* root) {
  if (root != NULL) {
    inOrderTraversal(root->left);
    printf("%d ", root->data);
    inOrderTraversal(root->right);
  }
}

int main() {
  Node* root = NULL;
  root = insert(root, 50);
  insert(root, 30);
  insert(root, 20);
  insert(root, 40);
  insert(root, 70);
  insert(root, 60);
  insert(root, 80);

  inOrderTraversal(root);

  return 0;
}

C++ - Binary Search Tree Implementation

Here's a similar implementation of a Binary Search Tree (BST) in C++: 

// C++ program for Binary Search Tree (BST)
#include <iostream>
using namespace std;

class Node {
  public:
    int data;
    Node* left;
    Node* right;
    Node(int val) {
      data = val;
      left = NULL;
      right = NULL;
    }
};

Node* insert(Node* root, int key) {
  if (root == NULL)
    return new Node(key);
  if (key < root->data)
    root->left = insert(root->left, key);
  else
    root->right = insert(root->right, key);
  return root;
}

void inOrderTraversal(Node* root) {
  if (root != NULL) {
    inOrderTraversal(root->left);
    cout << root->data << " ";
    inOrderTraversal(root->right);
  }
}

int main() {
  Node* root = insert(NULL, 50);
  insert(root, 30);
  insert(root, 20);
  insert(root, 40);
  insert(root, 70);
  insert(root, 60);
  insert(root, 80);

  inOrderTraversal(root);
  return 0;
}

Python - Binary Search Tree Implementation

Here's how you can implement a Binary Search Tree (BST) in Python: 

# Python program for Binary Search Tree (BST) implementation

class Node:
  def __init__(self, key):
    self.left = None
    self.right = None
    self.data = key

def insert(self, key):
  if key < self.data:
    if self.left is None:
      self.left = Node(key)
    else:
      self.left.insert(key)
  elif key > self.data:
    if self.right is None:
      self.right = Node(key)
    else:
      self.right.insert(key)

def inOrderTraversal(self):
  if self.left:
    self.left.inOrderTraversal()
  print(self.data, "", end"")
  if self.right:
    self.right.inOrderTraversal()

if __name__ == "__main__":
  root = Node(50)
  root.insert(30)
  root.insert(20)
  root.insert(40)
  root.insert(70)
  root.insert(60)
  root.insert(80)
  
  root.inOrderTraversal()
