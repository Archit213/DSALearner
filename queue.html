C - Queue Data Structure

C - Queue Implementation

In C, a queue can be implemented using arrays or linked lists. Below is an example of a simple queue using arrays: 

// C program to demonstrate a simple queue implementation
#include <stdio.h>
#define MAX 100

int queue[MAX];
int front = -1, rear = -1;

void enqueue(int element) {
  if (rear == MAX - 1) {
    // Queue is full
    return;
  }
  if (front == -1) {
    front = 0;
  }
  queue[++rear] = element;
}

int dequeue() {
  if (front == -1) {
    // Queue is empty
    return -1;
  }
  int element = queue[front++];
  if (front > rear) {
    front = rear = -1;
  }
  return element;
}

C++ - Queue Implementation

In C++, the queue can be implemented using the Standard Template Library (STL). Here's an example: 

// C++ program to demonstrate a simple queue implementation
#include <iostream>
#include <queue>

int main() {
  std::queue<int> q;
  q.push(10);
  q.push(20);
  q.push(30);

  while (!q.empty()) {
    std::cout << q.front() << " ";
    q.pop();
  }
  return 0;
}

Python - Queue Implementation

In Python, you can use the collections.deque class for an efficient queue implementation: 

# Python program to demonstrate a queue implementation
from collections import deque

queue = deque()
queue.append(10)
queue.append(20)
queue.append(30)

while queue:
    print(queue.popleft(), end=" ")
