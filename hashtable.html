C - Hash Table Data Structure

C - Hash Table Implementation

A hash table is a data structure that maps keys to values for efficient lookups. Here is a basic implementation in C: 

// C program to demonstrate a simple hash table implementation
#include <stdio.h>
#include <stdlib.h>

#define TABLE_SIZE 10

typedef struct Entry {
  int key;
  int value;
  struct Entry* next;
} Entry;

Entry* hash_table[TABLE_SIZE];

int hash(int key) {
  return key % TABLE_SIZE;
}

void insert(int key, int value) {
  int index = hash(key);
  Entry* new_entry = malloc(sizeof(Entry));
  new_entry->key = key;
  new_entry->value = value;
  new_entry->next = hash_table[index];
  hash_table[index] = new_entry;
}

int search(int key) {
  int index = hash(key);
  Entry* entry = hash_table[index];
  while (entry) {
    if (entry->key == key) {
      return entry->value;
    }
    entry = entry->next;
  }
  return -1; // Key not found
}

C++ - Hash Table Implementation

In C++, hash tables can be efficiently implemented using the STL unordered_map. Here's an example: 

// C++ program to demonstrate a simple hash table implementation
#include <iostream>
#include <unordered_map>

int main() {
  std::unordered_map<int, int> hash_table;
  hash_table[1] = 10;
  hash_table[2] = 20;
  hash_table[3] = 30;

  for (const auto& pair : hash_table) {
    std::cout << pair.first << " : " << pair.second << "\n";
  }

  return 0;
}

Python - Hash Table Implementation

In Python, hash tables are implemented using the built-in dict type. Here's a basic example: 

# Python program to demonstrate a simple hash table implementation
hash_table = {1: 10, 2: 20, 3: 30}

for key, value in hash_table.items():
    print(key, ":", value)
